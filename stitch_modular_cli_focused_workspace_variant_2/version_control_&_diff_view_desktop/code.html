/**
* Process raw data input
*/
export const processData = (data: any[]) => {
if (!data || data.length === 0) return [];
// Optimized processing with validation
return data.reduce((acc, item) => {
if (item.active) {
acc.push({
id: item.id,
value: Number(item.val) || 0,
timestamp: new Date().toISOString()
});
}
return acc;
}, []);
}
}
