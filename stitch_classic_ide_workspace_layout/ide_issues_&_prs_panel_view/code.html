import { AuthService } from './auth.service'; import { User } from './models/user'; /** * Authenticates user credentials against the API * Fixes issue #204 where null tokens caused crash */ export class LoginController { constructor(private auth: AuthService) {} async handleLogin(req: Request, res: Response) { try { const { email, password } = req.body; // Validate input before processing if (!email || !password) { throw new Error('Missing credentials'); } const token = await this.auth.login(email, password); return res.json({ success: true, token }); } catch (error) { console.error('Login failed', error); return res.status(401).send({ error: 'Unauthorized' }); } } }
